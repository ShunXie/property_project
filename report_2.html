all_val = 
  hp_three_yr %>%
  filter(City_name=="Chappaqua" | City_name=="Scarsdale") %>% 
  pivot_longer(

    names_to = "date"

  mutate(date_val = as.Date(as.numeric(date),origin="1899-12-30"),
